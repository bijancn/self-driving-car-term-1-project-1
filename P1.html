import math

def grayscale(img):
    """Applies the Grayscale transform
    This will return an image with only one color channel
    but NOTE: to see the returned image as grayscale
    (assuming your grayscaled image is called 'gray')
    you should call plt.imshow(gray, cmap='gray')"""
    return cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
    # Or use BGR2GRAY if you read an image with cv2.imread()
    # return cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    
def canny(img, low_threshold, high_threshold):
    """Applies the Canny transform"""
    return cv2.Canny(img, low_threshold, high_threshold)

def gaussian_blur(img, kernel_size):
    """Applies a Gaussian Noise kernel"""
    return cv2.GaussianBlur(img, (kernel_size, kernel_size), 0)

def region_of_interest(img, vertices):
    """
    Applies an image mask.
    
    Only keeps the region of the image defined by the polygon
    formed from `vertices`. The rest of the image is set to black.
    """
    #defining a blank mask to start with
    mask = np.zeros_like(img)   
    
    #defining a 3 channel or 1 channel color to fill the mask with depending on the input image
    if len(img.shape) > 2:
        channel_count = img.shape[2]  # i.e. 3 or 4 depending on your image
        ignore_mask_color = (255,) * channel_count
    else:
        ignore_mask_color = 255
        
    #filling pixels inside the polygon defined by "vertices" with the fill color    
    cv2.fillPoly(mask, vertices, ignore_mask_color)
    
    #returning the image only where mask pixels are nonzero
    masked_image = cv2.bitwise_and(img, mask)
    return masked_image

def get_normal_form(x1, y1, x2, y2):
    # parametrize a line by y = m * x + c
    m = (y2 - y1) / (x2 - x1)
    c = y1 - m * x1
    return (m, c)

def extrapolate_line(m, c, yshape):
    # we want to draw from bottom to horizon
    y_bottom = int(yshape)
    x_bottom = int((y_bottom - c) / m)
    y_top = int(yshape / 2.0 + yshape / 10.0)     # could be parametrized if needed
    x_top = int((y_top - c) / m)
    return (x_bottom, y_bottom, x_top, y_top)

def stable_mean(some_list, last_average):
    average = np.average(some_list)
    list_max = np.max(some_list)
    list_min = np.min(some_list)
    # filter out outliers
    stable = [elem for elem in some_list if (abs(elem - average) / abs(list_max - list_min) < 0.8)]
    stable_average = np.average(stable)
    if np.isnan(stable_average):
        result = average # fall back to average if filtering did not leave anything
    else:
        result = stable_average
    if last_average is not None:
        return np.average([result, last_average])
    else:
        return result
    
def draw_lines(img, lines, yshape,
               last_results={},
               left_color=[255, 0, 0],
               right_color=[0, 255, 0],
               thickness=5,
               show_individual_lines=False):
    left_lines, right_lines = [], []
    left_m, right_m = [], []
    left_c, right_c = [], []
    for line in lines:
        for x1, y1, x2, y2 in line:
            (m, c) = get_normal_form(x1, y1, x2, y2)
            if abs(m) > 0.1 and not np.isinf(c):
                (x_bottom, y_bottom, x_top, y_top) = extrapolate_line(m, c, yshape)
                if (m > 0):
                    left_lines.append((x_bottom, y_bottom, x_top, y_top))
                    left_m.append(m)
                    left_c.append(c)
                else:
                    right_lines.append((x_bottom, y_bottom, x_top, y_top))
                    right_m.append(m)
                    right_c.append(c)
    result_dict = {}
    if (len(left_lines) > 0):
        left_m_av = stable_mean(left_m, last_results.get('left_m', None))
        left_c_av = stable_mean(left_c, last_results.get('left_c', None))
        (x1, y1, x2, y2) = extrapolate_line(left_m_av, left_c_av, yshape)
        cv2.line(img, (x1, y1), (x2, y2), left_color, thickness)
        result_dict.update({'left_m': left_m_av, 'left_c': left_c_av})
    if (len(right_lines) > 0):
        right_m_av = stable_mean(right_m, last_results.get('right_m', None))
        right_c_av = stable_mean(right_c, last_results.get('right_c', None))
        (x1, y1, x2, y2) = extrapolate_line(right_m_av, right_c_av, yshape)
        cv2.line(img, (x1, y1), (x2, y2), right_color, thickness)
        result_dict.update({'right_m': right_m_av, 'right_c': right_c_av})
    if (show_individual_lines):
        for x1, y1, x2, y2 in left_lines:
            cv2.line(img, (x1, y1), (x2, y2), left_color, int(thickness/2))
        for x1, y1, x2, y2 in right_lines:
            cv2.line(img, (x1, y1), (x2, y2), right_color, int(thickness/2))
    return result_dict

def hough_lines(img, rho, theta, threshold, min_line_len, max_line_gap, yshape, last_results={}):
    """
    `img` should be the output of a Canny transform.
        
    Returns an image with hough lines drawn.
    """
    lines = cv2.HoughLinesP(img, rho, theta, threshold, np.array([]), minLineLength=min_line_len, maxLineGap=max_line_gap)
    line_img = np.zeros((img.shape[0], img.shape[1], 3), dtype=np.uint8)
    results = draw_lines(line_img, lines, yshape, last_results)
    return line_img, results

def weighted_img(img, initial_img, α=0.8, β=1., γ=0.):
    """
    `img` is the output of the hough_lines(), An image with lines drawn on it.
    Should be a blank image (all black) with lines drawn on it.
    
    `initial_img` should be the image before any processing.
    
    The result image is computed as follows:
    
    initial_img * α + img * β + γ
    NOTE: initial_img and img must be the same shape!
    """
    return cv2.addWeighted(initial_img, α, img, β, γ)

Build the pipeline on the given test image¶

In [4]:
gauss_kernel_size = 5
canny_low_threshold = 50
canny_high_threshold = 150
hough_rho = 2
hough_theta = np.pi / 180
hough_threshold = 15
hough_min_line_len = 40
hough_max_line_gap = 20

def pipeline(img, last_result={}):
    gray_img = grayscale(img)
    blurred_img = cv2.GaussianBlur(gray_img,(gauss_kernel_size, gauss_kernel_size),0)
    canny_img = canny(blurred_img, canny_low_threshold, canny_high_threshold)
    yshape, xshape, _ = img.shape
    vertices = np.array([[(0,yshape),
                          (xshape / 2 - xshape / 20, yshape / 2 + yshape / 10),
                          (xshape / 2 + xshape / 20, yshape / 2 + yshape / 10),
                          (xshape,yshape)]], dtype=np.int32)
    cropped_img = region_of_interest(canny_img, vertices)
    houghed_img, results = hough_lines(cropped_img,
                              hough_rho,
                              hough_theta, 
                              hough_threshold, 
                              hough_min_line_len, 
                              hough_max_line_gap,
                              yshape,
                              last_result
                             )
    weighted = weighted_img(houghed_img, img, α=0.8, β=1., γ=0.)
    return weighted, results

result, _ = pipeline(image)
